# Các lệnh CLI thường dùng trong Neutron

## 1. Chuẩn bị

```bash
# CLI
pip install python-openstackclient python-neutronclient

# Source admin
source /root/admin-openrc

# Kiểm tra
openstack --version
openstack network agent list
```

> Lưu ý: Từ Pike trở đi, dùng `openstack network/subnet/port/router/...` thay vì `neutron` (legacy nhưng vẫn dùng được).

## 2. Network

```bash
# Liệt kê
openstack network list [--external] [--internal] [--share]
openstack network show <net>

# Tạo network tự phục vụ (project network - Geneve/VXLAN)
openstack network create private

# Tạo provider network (flat - admin only)
openstack network create --share --provider-network-type flat \
  --provider-physical-network physnet1 ext-net

# Tạo provider VLAN network
openstack network create --share \
  --provider-network-type vlan \
  --provider-physical-network physnet1 \
  --provider-segment 100 vlan-100

# Tạo external network (cho floating IP)
openstack network create --external --share \
  --provider-network-type flat \
  --provider-physical-network physnet1 \
  public

# Set/Unset
openstack network set --enable <net>          # admin_state_up = True
openstack network set --disable <net>
openstack network set --description "desc" <net>
openstack network set --share <net>
openstack network set --no-share <net>

# Xóa
openstack network delete <net>
```

## 3. Subnet

```bash
# Liệt kê
openstack subnet list [--network <net>]
openstack subnet show <subnet>

# Tạo subnet với DHCP
openstack subnet create private-subnet \
  --network private \
  --subnet-range 192.168.10.0/24 \
  --gateway 192.168.10.1 \
  --dns-nameserver 8.8.8.8 \
  --dns-nameserver 1.1.1.1 \
  --allocation-pool start=192.168.10.10,end=192.168.10.200

# Subnet không DHCP (external network)
openstack subnet create ext-subnet \
  --network ext-net \
  --no-dhcp \
  --subnet-range 192.168.1.0/24 \
  --gateway 192.168.1.1 \
  --allocation-pool start=192.168.1.100,end=192.168.1.200

# Subnet IPv6
openstack subnet create v6-subnet \
  --network private \
  --subnet-range 2001:db8::/64 \
  --ip-version 6 \
  --ipv6-ra-mode slaac \
  --ipv6-address-mode slaac

# Set
openstack subnet set --gateway <ip> <subnet>
openstack subnet set --dns-nameserver 8.8.8.8 <subnet>
openstack subnet set --allocation-pool start=...,end=... <subnet>
openstack subnet set --host-route destination=10.0.0.0/24,gateway=192.168.10.5 <subnet>

# Xóa
openstack subnet delete <subnet>
```

## 4. Subnet Pool

```bash
# Tạo pool
openstack subnet pool create --pool-prefix 10.0.0.0/16 \
  --default-prefix-length 24 mypool

# Liệt kê
openstack subnet pool list
openstack subnet pool show mypool

# Tạo subnet từ pool
openstack subnet create --subnet-pool mypool --prefix-length 24 \
  --network private auto-subnet

# Xóa
openstack subnet pool delete mypool
```

## 5. Port

```bash
# Liệt kê
openstack port list [--device-owner compute:nova]
openstack port show <port>

# Tạo port với IP cố định
openstack port create --network private \
  --fixed-ip subnet=private-subnet,ip-address=192.168.10.50 \
  --security-group default \
  my-port

# Tạo port không security (cho VNF)
openstack port create --network private \
  --disable-port-security \
  --no-security-group nfv-port

# Tạo port với MAC cố định
openstack port create --network private --mac-address fa:16:3e:11:22:33 fixed-mac

# Allowed-address-pairs (cho HA/VIP)
openstack port set --allowed-address \
  ip-address=192.168.10.99 <port>

# Disable port security (toàn bộ filter)
openstack port set --disable-port-security <port>

# Attach port vào VM (Nova command)
openstack server add port <vm> <port>
openstack server remove port <vm> <port>

# Xóa
openstack port delete <port>
```

## 6. Router

```bash
# Liệt kê
openstack router list
openstack router show <router>

# Tạo router
openstack router create my-router

# Set external gateway
openstack router set --external-gateway public my-router

# Thêm interface (nối router vào tenant subnet)
openstack router add subnet my-router private-subnet

# Hoặc thêm port cụ thể
openstack router add port my-router <port-id>

# Xóa interface
openstack router remove subnet my-router private-subnet
openstack router remove port my-router <port-id>

# Bỏ external gateway
openstack router unset --external-gateway my-router

# HA router
openstack router create --ha ha-router

# DVR
openstack router create --distributed dvr-router
openstack router set --distributed <router>      # Chuyển sang DVR (chỉ admin, cần stop trước)
openstack router set --no-ha <router>
openstack router set --no-distributed <router>

# Routes (static route)
openstack router set --route destination=10.0.0.0/24,gateway=192.168.10.5 my-router
openstack router unset --route destination=... my-router

# Xóa router
openstack router delete my-router
```

## 7. Floating IP

```bash
# Liệt kê
openstack floating ip list
openstack floating ip show <fip>

# Tạo (allocate từ external network)
openstack floating ip create public
openstack floating ip create --floating-ip-address 192.168.1.150 public

# Associate vào VM
openstack server add floating ip <vm> <fip-address>

# Hoặc associate vào port cụ thể
openstack floating ip set --port <port-id> <fip>

# Disassociate
openstack server remove floating ip <vm> <fip>
openstack floating ip unset --port <fip>

# Xóa (release)
openstack floating ip delete <fip>
```

## 8. Security Group & Rules

```bash
# Liệt kê
openstack security group list
openstack security group show default

# Tạo group
openstack security group create web --description "Web servers"

# Thêm rule SSH
openstack security group rule create --proto tcp --dst-port 22 \
  --remote-ip 0.0.0.0/0 web

# Thêm rule HTTP/HTTPS
openstack security group rule create --proto tcp --dst-port 80 --remote-ip 0.0.0.0/0 web
openstack security group rule create --proto tcp --dst-port 443 --remote-ip 0.0.0.0/0 web

# ICMP (ping)
openstack security group rule create --proto icmp --remote-ip 0.0.0.0/0 web

# Port range
openstack security group rule create --proto tcp --dst-port 8000:8100 --remote-ip 10.0.0.0/8 web

# Egress (mặc định allow all - hiếm khi cần set)
openstack security group rule create --egress --proto tcp --dst-port 443 web

# Rule cho phép từ security group khác (thay vì IP)
openstack security group rule create --proto tcp --dst-port 3306 \
  --remote-group web db

# Liệt kê rule trong group
openstack security group rule list web

# Xóa rule
openstack security group rule delete <rule-id>

# Xóa group
openstack security group delete web

# Gán/bỏ SG khỏi VM/port
openstack server add security group <vm> web
openstack server remove security group <vm> web
openstack port set --security-group web <port>
openstack port unset --security-group web <port>
```

## 9. QoS

```bash
# Tạo QoS policy
openstack network qos policy create bw-policy

# Thêm rule giới hạn bandwidth
openstack network qos rule create --type bandwidth-limit \
  --max-kbps 10000 --max-burst-kbits 1000 \
  --egress bw-policy

# DSCP marking
openstack network qos rule create --type dscp-marking \
  --dscp-mark 26 bw-policy

# Min bandwidth (đảm bảo BW tối thiểu)
openstack network qos rule create --type minimum-bandwidth \
  --min-kbps 5000 --egress bw-policy

# Gán policy vào network/port
openstack network set --qos-policy bw-policy private
openstack port set --qos-policy bw-policy <port>

# Liệt kê / xem
openstack network qos policy list
openstack network qos policy show bw-policy
openstack network qos rule list bw-policy
```

## 10. Trunk Port (VLAN trunking cho K8s/VNF)

```bash
# Tạo parent port
openstack port create --network private parent-port

# Tạo subport
openstack port create --network vlan-100 sub-port-100

# Tạo trunk
openstack network trunk create my-trunk --parent-port parent-port

# Thêm subport vào trunk với segmentation ID
openstack network trunk set my-trunk \
  --subport port=sub-port-100,segmentation-type=vlan,segmentation-id=100

# Liệt kê
openstack network trunk list
openstack network trunk show my-trunk

# Xóa subport
openstack network trunk unset my-trunk --subport sub-port-100

# Xóa trunk
openstack network trunk delete my-trunk
```

## 11. Address Scope & BGP

```bash
# Address scope (gom các subnet pool có routable IP chung)
openstack address scope create --ip-version 4 --share public-scope

# Tạo subnet pool gắn vào scope
openstack subnet pool create \
  --pool-prefix 203.0.113.0/24 \
  --address-scope public-scope \
  public-pool

# BGP (cần neutron-dynamic-routing plugin)
openstack bgp speaker create --local-as 65000 my-speaker
openstack bgp peer create --peer-ip 10.0.0.1 --remote-as 65001 my-peer
openstack bgp speaker add peer my-speaker my-peer
```

## 14. Tools cấp hệ thống (debug)

```bash
# Trên Network/Compute node
sudo ip netns                                 # Liệt kê namespace
sudo ip netns exec qrouter-<id> ip addr       # Xem IP trong router namespace
sudo ip netns exec qdhcp-<id> ip route        # Xem route trong DHCP namespace

# OVS
sudo ovs-vsctl show
sudo ovs-vsctl list-br
sudo ovs-vsctl list-ports br-int
sudo ovs-ofctl dump-flows br-int

# OVN
sudo ovn-nbctl show                           # Logical view
sudo ovn-sbctl show                           # Physical binding
sudo ovn-sbctl list chassis
sudo ovn-trace <datapath> '<microflow>'       # Debug flow

# Iptables (legacy security group)
sudo iptables -L -n -v
sudo iptables -t nat -L -n -v

# Bridge / TAP
sudo ip link show
brctl show
sudo tcpdump -i tap<xxxxx> -nn
```

## 15. Trace + Debug API

```bash
# Bật debug client
openstack --debug network create test 2>&1 | grep -i "x-openstack-request-id"

# Trace request-id qua log
REQ=req-xxxxxxxx
grep "$REQ" /var/log/neutron/*.log

# Xem fault của port/router
openstack port show <port> -c status -c binding_vif_type
openstack router show <router> -c status
```

## 16. Quick reference – Tạo network end-to-end

```bash
# Network + Subnet
openstack network create private
openstack subnet create private-subnet --network private \
  --subnet-range 192.168.10.0/24 --gateway 192.168.10.1 \
  --dns-nameserver 8.8.8.8

# External + Router
openstack network create --external --share \
  --provider-network-type flat --provider-physical-network physnet1 public
openstack subnet create public-subnet --network public --no-dhcp \
  --subnet-range 192.168.1.0/24 --gateway 192.168.1.1 \
  --allocation-pool start=192.168.1.100,end=192.168.1.200

openstack router create my-router
openstack router set --external-gateway public my-router
openstack router add subnet my-router private-subnet

# Security Group
openstack security group create web
openstack security group rule create --proto icmp web
openstack security group rule create --proto tcp --dst-port 22 web

# VM + Floating IP
openstack server create vm1 --image cirros --flavor m1.tiny \
  --network private --security-group web

FIP=$(openstack floating ip create public -f value -c floating_ip_address)
openstack server add floating ip vm1 $FIP
ping $FIP
```
